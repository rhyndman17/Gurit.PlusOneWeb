CREATE OR ALTER PROC [dbo].[hmlPlusOneCreateNonPOInvoice]
    @BatchID char(15) = 'PLUSONE'
AS
BEGIN
    SET NOCOUNT ON;

    DECLARE
        @InvoiceHeaderID int,
        @ErrorMessage nvarchar(4000);

    DECLARE curPlusOneReadyInvoice CURSOR LOCAL FAST_FORWARD FOR
        SELECT InvoiceHeaderID
        FROM dbo.hmlPlusOneInvoiceHeader
        WHERE Status = 0
        ORDER BY DocumentDate, InvoiceHeaderID;

    OPEN curPlusOneReadyInvoice;

    FETCH NEXT FROM curPlusOneReadyInvoice
    INTO @InvoiceHeaderID;

    WHILE @@FETCH_STATUS = 0
    BEGIN
        BEGIN TRY
            EXEC dbo.hmlPlusOneProcessInvoice
                @InvoiceHeaderID = @InvoiceHeaderID,
                @BatchID = @BatchID;
        END TRY
        BEGIN CATCH
            SET @ErrorMessage = ERROR_MESSAGE();
            PRINT 'PlusOne invoice header '
                + CAST(@InvoiceHeaderID AS varchar(12))
                + ' failed: '
                + @ErrorMessage;
        END CATCH;

        FETCH NEXT FROM curPlusOneReadyInvoice
        INTO @InvoiceHeaderID;
    END;

    CLOSE curPlusOneReadyInvoice;
    DEALLOCATE curPlusOneReadyInvoice;
END
GO
