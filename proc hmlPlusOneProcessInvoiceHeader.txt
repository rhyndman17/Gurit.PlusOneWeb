CREATE OR ALTER PROC [dbo].[hmlPlusOneProcessInvoiceHeader]
    @InvoiceHeaderID int,
    @BatchID char(15) = 'PLUSONE'
AS
BEGIN
    SET NOCOUNT ON;

    DECLARE
        @O_iErrorState int,
        @oErrString varchar(255),
        @sErrMessage varchar(4000),
        @sVendorID char(15),
        @sDocNumber char(21),
        @dtDocDate datetime,
        @dtAccountingDate datetime,
        @iTransType int,
        @cDocSubtotal numeric(18, 2),
        @cTaxAmount numeric(18, 2),
        @cDocAmount numeric(18, 2),
        @sFileName char(200),
        @sNoteText char(255),
        @iNoteIndex int,
        @sVoucherNumber char(17),
        @sBatchID char(15),
        @sTaxSched char(15),
        @sCurrencyID char(10),
        @cCurrencyRate numeric(18, 5),
        @iCurrencyIndex int,
        @sRateTypeID char(15),
        @dtExchangeDate datetime,
        @sExchangeTableID char(15),
        @cOrigDistAmount numeric(18, 2),
        @iExpAccountIndex int,
        @noteExists int,
        @CurrentLineID int,
        @CurrentSourceLineNo int,
        @sExpAccount char(50),
        @cDistAmount numeric(18, 2),
        @sLineDescription char(30),
        @HeaderStatus int,
        @ErrorMessage nvarchar(4000);

    SET @O_iErrorState = 0;
    SET @oErrString = '';
    SET @sErrMessage = '';
    SET @CurrentLineID = NULL;
    SET @CurrentSourceLineNo = NULL;
    SET @sBatchID = @BatchID;
    SET @iTransType = 1;

    SELECT
        @HeaderStatus = h.Status,
        @sVendorID = h.SupplierID,
        @sDocNumber = RTRIM(UPPER(h.DocumentNo)),
        @dtDocDate = h.DocumentDate,
        @dtAccountingDate = h.AccountingValueDate,
        @sNoteText = LEFT(ISNULL(h.ImageURL, ''), 255),
        @sFileName = h.FileName
    FROM dbo.hmlPlusOneInvoiceHeader h WITH (UPDLOCK, HOLDLOCK)
    WHERE h.InvoiceHeaderID = @InvoiceHeaderID;

    IF @sVendorID IS NULL
    BEGIN
        RAISERROR('PlusOne invoice header was not found.', 16, 1);
        RETURN;
    END;

    IF @HeaderStatus = 1
    BEGIN
        RAISERROR('PlusOne invoice header is already being processed.', 16, 1);
        RETURN;
    END;

    IF @HeaderStatus = 2
    BEGIN
        RAISERROR('PlusOne invoice header has already been processed.', 16, 1);
        RETURN;
    END;

    UPDATE dbo.hmlPlusOneInvoiceHeader
    SET
        Status = 1,
        Processed = NULL,
        GPVoucherNumber = NULL,
        GPBatchID = NULL,
        HeaderMessage = NULL,
        LastUpdatedDateTime = GETDATE()
    WHERE InvoiceHeaderID = @InvoiceHeaderID;

    UPDATE dbo.hmlPlusOneInvoiceLine
    SET
        Status = 0,
        Processed = NULL,
        ErrorState = NULL,
        LineMessage = NULL,
        LastUpdatedDateTime = GETDATE()
    WHERE InvoiceHeaderID = @InvoiceHeaderID;

    BEGIN TRY
        IF NOT EXISTS (
            SELECT 1
            FROM dbo.hmlPlusOneInvoiceLine
            WHERE InvoiceHeaderID = @InvoiceHeaderID
        )
        BEGIN
            SET @sErrMessage = 'No PlusOne lines were found for the selected invoice.';
            RAISERROR(@sErrMessage, 16, 1);
        END;

        IF EXISTS (
            SELECT 1 FROM PM10000 WHERE DOCNUMBR = @sDocNumber AND VENDORID = @sVendorID
            UNION ALL
            SELECT 1 FROM PM20000 WHERE DOCNUMBR = @sDocNumber AND VENDORID = @sVendorID
            UNION ALL
            SELECT 1 FROM PM30200 WHERE DOCNUMBR = @sDocNumber AND VENDORID = @sVendorID
        )
        BEGIN
            SET @sErrMessage = 'Duplicate document No.';
            RAISERROR(@sErrMessage, 16, 1);
        END;

        IF NOT EXISTS (
            SELECT 1
            FROM PM00200
            WHERE VENDORID = @sVendorID
        )
        BEGIN
            SET @sErrMessage = 'Vendor was not found in PM00200: ' + RTRIM(@sVendorID);
            RAISERROR(@sErrMessage, 16, 1);
        END;

        SELECT
            @cDocSubtotal = SUM(ISNULL(LineValueExclGST, 0)),
            @cTaxAmount = SUM(ISNULL(LineValueGST, 0))
        FROM dbo.hmlPlusOneInvoiceLine
        WHERE InvoiceHeaderID = @InvoiceHeaderID;

        SET @cDocSubtotal = ISNULL(@cDocSubtotal, 0);
        SET @cTaxAmount = ISNULL(@cTaxAmount, 0);
        SET @cDocAmount = @cDocSubtotal + @cTaxAmount;

        SELECT @sCurrencyID = CURNCYID
        FROM PM00200
        WHERE VENDORID = @sVendorID;

        IF NULLIF(RTRIM(ISNULL(@sCurrencyID, '')), '') IS NULL
        BEGIN
            SET @sErrMessage = 'Vendor currency is blank for vendor ' + RTRIM(@sVendorID) + '.';
            RAISERROR(@sErrMessage, 16, 1);
        END;

        SET @sTaxSched = CASE
            WHEN @sCurrencyID = 'NZD' THEN 'PUREX15%'
            WHEN @sCurrencyID = 'AUD' THEN 'PUREX10%'
            ELSE 'PURZR'
        END;

        BEGIN TRAN;

        SET @sVoucherNumber = '';
        EXECUTE taGetPMNextVoucherNumber
            @sVoucherNumber OUTPUT,
            0,
            @O_iErrorState OUTPUT;

        IF @O_iErrorState <> 0
        BEGIN
            SET @sErrMessage = ISNULL(
                (
                    SELECT ErrorDesc + ' : APVCHR : ' + RTRIM(@sDocNumber) + ' - ' + RTRIM(@sVendorID)
                    FROM DYNAMICS..taErrorCode
                    WHERE ErrorCode = @O_iErrorState
                ),
                'Unable to retrieve the next PM voucher number.'
            );
            RAISERROR(@sErrMessage, 16, 1);
        END;

        EXECUTE taPMTransactionTaxInsert
            @I_vVENDORID = @sVendorID,
            @I_vVCHRNMBR = @sVoucherNumber,
            @I_vDOCTYPE = @iTransType,
            @I_vBACHNUMB = @sBatchID,
            @I_vTAXDTLID = @sTaxSched,
            @I_vPCTAXAMT = @cDocSubtotal,
            @I_vTDTTXPUR = @cDocSubtotal,
            @I_vTXDTTPUR = @cDocSubtotal,
            @I_vTAXAMNT = @cTaxAmount,
            @O_iErrorState = @O_iErrorState OUTPUT,
            @oErrString = @oErrString OUTPUT;

        IF @O_iErrorState <> 0
        BEGIN
            SET @sErrMessage = ISNULL(
                (
                    SELECT ErrorDesc + ' : APTAX : ' + RTRIM(@sDocNumber) + ' - ' + RTRIM(@sVendorID)
                    FROM DYNAMICS..taErrorCode
                    WHERE ErrorCode = @O_iErrorState
                ),
                'Unable to create the PM tax record.'
            );
            RAISERROR(@sErrMessage, 16, 1);
        END;

        EXECUTE taPMTransactionInsert
            @I_vBACHNUMB = @sBatchID,
            @I_vVCHNUMWK = @sVoucherNumber,
            @I_vVENDORID = @sVendorID,
            @I_vDOCNUMBR = @sDocNumber,
            @I_vDOCTYPE = @iTransType,
            @I_vPRCHAMNT = @cDocSubtotal,
            @I_vTAXAMNT = @cTaxAmount,
            @I_vDOCAMNT = @cDocAmount,
            @I_vCHRGAMNT = @cDocAmount,
            @I_vDOCDATE = @dtDocDate,
            @I_vTAXSCHID = @sTaxSched,
            @I_vPCHSCHID = @sTaxSched,
            @I_vFRTSCHID = @sTaxSched,
            @I_vMSCSCHID = @sTaxSched,
            @I_vCREATEDIST = 1,
            @O_iErrorState = @O_iErrorState OUTPUT,
            @oErrString = @oErrString OUTPUT;

        IF @O_iErrorState <> 0
        BEGIN
            SET @sErrMessage = ISNULL(
                (
                    SELECT ErrorDesc + ' : APHDR : ' + RTRIM(@sDocNumber) + ' - ' + RTRIM(@sVendorID)
                    FROM DYNAMICS..taErrorCode
                    WHERE ErrorCode = @O_iErrorState
                ),
                'Unable to create the PM invoice header.'
            );
            RAISERROR(@sErrMessage, 16, 1);
        END;

        SET @iNoteIndex = (SELECT NOTEINDX FROM PM10000 WHERE VCHNUMWK = @sVoucherNumber);
        SET @noteExists = (SELECT COUNT(*) FROM SY03900 WHERE NOTEINDX = @iNoteIndex);

        IF @iNoteIndex > 0
        BEGIN
            IF @noteExists = 0
            BEGIN
                INSERT INTO SY03900
                SELECT @iNoteIndex, ISNULL(@dtAccountingDate, @dtDocDate), '', @sNoteText;
            END;

            IF @noteExists > 0
            BEGIN
                UPDATE SY03900
                SET TXTFIELD = @sNoteText
                WHERE NOTEINDX = @iNoteIndex;
            END;
        END;

        DELETE PM10100
        WHERE VCHRNMBR = @sVoucherNumber
          AND DISTTYPE = 6;

        DECLARE curBuildDistributions CURSOR LOCAL FAST_FORWARD FOR
            SELECT
                InvoiceLineID,
                SourceLineNo,
                LedgerCode,
                LineValueExclGST,
                LineDescription
            FROM dbo.hmlPlusOneInvoiceLine
            WHERE InvoiceHeaderID = @InvoiceHeaderID
            ORDER BY SourceLineNo, InvoiceLineID;

        OPEN curBuildDistributions;

        FETCH NEXT FROM curBuildDistributions
        INTO @CurrentLineID, @CurrentSourceLineNo, @sExpAccount, @cDistAmount, @sLineDescription;

        WHILE @@FETCH_STATUS = 0
        BEGIN
            SET @iExpAccountIndex = (
                SELECT ACTINDX
                FROM GL00105
                WHERE ACTNUMST = @sExpAccount
            );

            IF ISNULL(@iExpAccountIndex, 0) = 0
            BEGIN
                SET @sErrMessage = 'Line ' + CAST(@CurrentSourceLineNo AS varchar(12))
                    + ': GL account was not found for LedgerCode ' + RTRIM(@sExpAccount) + '.';
                RAISERROR(@sErrMessage, 16, 1);
            END;

            IF @sCurrencyID <> 'NZD'
            BEGIN
                SELECT
                    @cCurrencyRate = XCHGRATE,
                    @iCurrencyIndex = CURRNIDX,
                    @sRateTypeID = RATETPID,
                    @dtExchangeDate = EXCHDATE,
                    @sExchangeTableID = EXGTBLID
                FROM MC020103
                WHERE VCHRNMBR = @sVoucherNumber;

                IF ISNULL(@cCurrencyRate, 0) = 0
                BEGIN
                    SET @sErrMessage = 'Line ' + CAST(@CurrentSourceLineNo AS varchar(12))
                        + ': Missing or zero currency rate for voucher ' + RTRIM(@sVoucherNumber) + '.';
                    RAISERROR(@sErrMessage, 16, 1);
                END;

                SET @cOrigDistAmount = ROUND(@cDistAmount / @cCurrencyRate, 2);

                EXECUTE taPMDistributionInsert
                    @I_vDOCTYPE = 1,
                    @I_vVCHNUMWK = @sVoucherNumber,
                    @I_vVENDORID = @sVendorID,
                    @I_vDistRef = @sLineDescription,
                    @I_vACTINDX = @iExpAccountIndex,
                    @I_vCURRNIDX = @iCurrencyIndex,
                    @I_vCURNCYID = @sCurrencyID,
                    @I_vDISTTYPE = 6,
                    @I_vDEBITAMT = @cOrigDistAmount,
                    @I_vDECPLCUR = 2,
                    @I_vCRDTAMNT = 0,
                    @I_vTRXSORCE = '',
                    @I_vPTDUSRID = '',
                    @I_vRATETPID = @sRateTypeID,
                    @I_vXCHGRATE = @cCurrencyRate,
                    @I_vEXCHDATE = @dtExchangeDate,
                    @I_vEXPNDATE = '',
                    @I_vEXGTBLID = @sExchangeTableID,
                    @I_vTIME1 = '1900-01-01 08:00:00.000',
                    @I_vORDBTAMT = @cDistAmount,
                    @I_vORCRDAMT = 0,
                    @I_vRTCLCMTD = 1,
                    @O_iErrorState = @O_iErrorState OUTPUT,
                    @oErrString = @oErrString OUTPUT;

                IF @O_iErrorState <> 0
                BEGIN
                    SET @sErrMessage = ISNULL(
                        (
                            SELECT 'Line ' + CAST(@CurrentSourceLineNo AS varchar(12)) + ': '
                                + ErrorDesc + ' : APDIST : ' + RTRIM(@sVoucherNumber)
                            FROM DYNAMICS..taErrorCode
                            WHERE ErrorCode = @O_iErrorState
                        ),
                        'Line ' + CAST(@CurrentSourceLineNo AS varchar(12))
                            + ': Unable to create the PM distribution.'
                    );
                    RAISERROR(@sErrMessage, 16, 1);
                END;
            END;
            ELSE
            BEGIN
                EXECUTE taPMDistribution
                    @I_vDOCTYPE = 1,
                    @I_vVCHRNMBR = @sVoucherNumber,
                    @I_vVENDORID = @sVendorID,
                    @I_vDistRef = @sLineDescription,
                    @I_vACTINDX = @iExpAccountIndex,
                    @I_vDISTTYPE = 6,
                    @I_vDEBITAMT = @cDistAmount,
                    @I_vCRDTAMNT = 0,
                    @O_iErrorState = @O_iErrorState OUTPUT,
                    @oErrString = @oErrString OUTPUT;

                IF @O_iErrorState <> 0
                BEGIN
                    SET @sErrMessage = ISNULL(
                        (
                            SELECT 'Line ' + CAST(@CurrentSourceLineNo AS varchar(12)) + ': '
                                + ErrorDesc + ' : APDIST : ' + RTRIM(@sVoucherNumber)
                            FROM DYNAMICS..taErrorCode
                            WHERE ErrorCode = @O_iErrorState
                        ),
                        'Line ' + CAST(@CurrentSourceLineNo AS varchar(12))
                            + ': Unable to create the PM distribution.'
                    );
                    RAISERROR(@sErrMessage, 16, 1);
                END;
            END;

            FETCH NEXT FROM curBuildDistributions
            INTO @CurrentLineID, @CurrentSourceLineNo, @sExpAccount, @cDistAmount, @sLineDescription;
        END;

        CLOSE curBuildDistributions;
        DEALLOCATE curBuildDistributions;

        COMMIT TRAN;

        UPDATE dbo.hmlPlusOneInvoiceHeader
        SET
            Status = 2,
            Processed = GETDATE(),
            GPVoucherNumber = @sVoucherNumber,
            GPBatchID = @sBatchID,
            HeaderMessage = NULL,
            LastUpdatedDateTime = GETDATE()
        WHERE InvoiceHeaderID = @InvoiceHeaderID;

        UPDATE dbo.hmlPlusOneInvoiceLine
        SET
            Status = 2,
            Processed = GETDATE(),
            ErrorState = NULL,
            LineMessage = NULL,
            LastUpdatedDateTime = GETDATE()
        WHERE InvoiceHeaderID = @InvoiceHeaderID;

        INSERT INTO dbo.hmlPlusOneMessages (
            MessageDateTime,
            DetailDesc1,
            DetailDesc2,
            MessageID,
            MessageState,
            MessageString1,
            MessageString2
        )
        VALUES (
            GETDATE(),
            LEFT(RTRIM(@sDocNumber), 100),
            LEFT(RTRIM(@sVendorID), 100),
            'P1PROC',
            0,
            LEFT(RTRIM(@sFileName), 255),
            LEFT('Processed voucher ' + RTRIM(@sVoucherNumber), 255)
        );
    END TRY
    BEGIN CATCH
        IF CURSOR_STATUS('local', 'curBuildDistributions') > -1
        BEGIN
            CLOSE curBuildDistributions;
            DEALLOCATE curBuildDistributions;
        END
        ELSE IF CURSOR_STATUS('local', 'curBuildDistributions') = -1
        BEGIN
            DEALLOCATE curBuildDistributions;
        END;

        IF @@TRANCOUNT > 0
        BEGIN
            ROLLBACK TRAN;
        END;

        SET @ErrorMessage = COALESCE(NULLIF(@sErrMessage, ''), ERROR_MESSAGE());

        UPDATE dbo.hmlPlusOneInvoiceHeader
        SET
            Status = 9,
            Processed = NULL,
            GPVoucherNumber = NULL,
            GPBatchID = NULL,
            HeaderMessage = @ErrorMessage,
            LastUpdatedDateTime = GETDATE()
        WHERE InvoiceHeaderID = @InvoiceHeaderID;

        IF @CurrentLineID IS NOT NULL
        BEGIN
            UPDATE dbo.hmlPlusOneInvoiceLine
            SET
                Status = 9,
                Processed = NULL,
                ErrorState = NULLIF(@O_iErrorState, 0),
                LineMessage = @ErrorMessage,
                LastUpdatedDateTime = GETDATE()
            WHERE InvoiceLineID = @CurrentLineID;
        END;

        INSERT INTO dbo.hmlPlusOneMessages (
            MessageDateTime,
            DetailDesc1,
            DetailDesc2,
            MessageID,
            MessageState,
            MessageString1,
            MessageString2
        )
        VALUES (
            GETDATE(),
            LEFT(RTRIM(@sDocNumber), 100),
            LEFT(RTRIM(@sVendorID), 100),
            'PLUSONE',
            ISNULL(NULLIF(@O_iErrorState, 0), 9),
            LEFT(RTRIM(ISNULL(@sFileName, '')), 255),
            LEFT(@ErrorMessage, 255)
        );

        RAISERROR(@ErrorMessage, 16, 1);
    END CATCH;
END
